			add = {

				desc = court_position_diligent_trait
			}
		}
		else_if = {
			limit = {
				has_trait = lazy
			}
			add = { 
				value = -20
				desc = court_position_lazy_trait
			}
		}
		
		if = {
			limit = {
				has_trait = cynical
			}
			add = {
				value = 20

			}
		}
		
		if = {
			limit = {
				has_trait = overseer

		}

		if = {
			limit = {
				has_trait = scholar
			}
			add = {
				value = 15
				desc = court_position_scholar_trait
			}

	valid_position = { }

				is_courtier_of = scope:liege
				is_councillor_of = scope:liege
			}
